34// Now()
35//
36// Returns the current time, expressed as an `absl::Time` absolute time value.
37absl::Time Now();
38
39// GetCurrentTimeNanos()
40//
41// Returns the current time, expressed as a count of nanoseconds since the Unix
42// Epoch (https://en.wikipedia.org/wiki/Unix_time). Prefer `absl::Now()` instead
43// for all but the most performance-sensitive cases (i.e. when you are calling
44// this function hundreds of thousands of times per second).
45int64_t GetCurrentTimeNanos();
46
47// SleepFor()
48//
49// Sleeps for the specified duration, expressed as an `absl::Duration`.
50//
51// Notes:
52// * Signal interruptions will not reduce the sleep duration.
53// * Returns immediately when passed a nonpositive duration.
54void SleepFor(absl::Duration duration);

63// In some build configurations we pass --detect-odr-violations to the
64// gold linker. This causes it to flag weak symbol overrides as ODR
65// violations. Because ODR only applies to C++ and not C,
66// --detect-odr-violations ignores symbols not mangled with C++ names.
67// By changing our extension points to be extern "C", we dodge this
68// check.
69extern "C" {
70ABSL_DLL void ABSL_INTERNAL_C_SYMBOL(AbslInternalSleepFor)(
71 absl::Duration duration);
72} // extern "C"
73
74inline void absl::SleepFor(absl::Duration duration) {
75 ABSL_INTERNAL_C_SYMBOL(AbslInternalSleepFor)(duration);
76}
