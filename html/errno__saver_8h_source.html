26// `ErrnoSaver` captures the value of `errno` upon construction and restores it
27// upon deletion. It is used in low-level code and must be super fast. Do not
28// add instrumentation, even in debug modes.
29class ErrnoSaver {
30 public:
31 ErrnoSaver() : saved_errno_(errno) {}
32 ~ErrnoSaver() { errno = saved_errno_; }
33 int operator()() const { return saved_errno_; }
34
35 private:
36 const int saved_errno_;
37};
