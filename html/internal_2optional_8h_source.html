44// This class stores the data in optional<T>.
45// It is specialized based on whether T is trivially destructible.
46// This is the specialization for non trivially destructible type.
47template <typename T, bool unused = std::is_trivially_destructible<T>::value>
48class optional_data_dtor_base {
49 struct dummy_type {
50 static_assert(sizeof(T) % sizeof(empty_struct) == 0, "");
51 // Use an array to avoid GCC 6 placement-new warning.
52 empty_struct data[sizeof(T) / sizeof(empty_struct)];
53 };
54
55 protected:
56 // Whether there is data or not.
57 bool engaged_;
58 // Data storage
59 union {
60 T data_;
61 dummy_type dummy_;
62 };
63
64 void destruct() noexcept {
65 if (engaged_) {
66 // `data_` must be initialized if `engaged_` is true.
67#if ABSL_INTERNAL_HAVE_MIN_GNUC_VERSION(12, 0)
68#pragma GCC diagnostic push
69#pragma GCC diagnostic ignored "-Wmaybe-uninitialized"
70#endif
71 data_.~T();
72#if ABSL_INTERNAL_HAVE_MIN_GNUC_VERSION(12, 0)
73#pragma GCC diagnostic pop
74#endif
75 engaged_ = false;
76 }
77 }
78
79 // dummy_ must be initialized for constexpr constructor.
80 constexpr optional_data_dtor_base() noexcept : engaged_(false), dummy_{{}} {}
81
82 template <typename... Args>
83 constexpr explicit optional_data_dtor_base(in_place_t, Args&&... args)
84 : engaged_(true), data_(std::forward<Args>(args)...) {}
85
86 ~optional_data_dtor_base() { destruct(); }
87};

116template <typename T>
117class optional_data_base : public optional_data_dtor_base<T> {
118 protected:
119 using base = optional_data_dtor_base<T>;
120 using base::base;
121
122 template <typename... Args>
123 void construct(Args&&... args) {
124 // Use dummy_'s address to work around casting cv-qualified T* to void*.
125 ::new (static_cast<void*>(&this->dummy_)) T(std::forward<Args>(args)...);
126 this->engaged_ = true;
127 }
128
129 template <typename U>
130 void assign(U&& u) {
131 if (this->engaged_) {
132 this->data_ = std::forward<U>(u);
133 } else {
134 construct(std::forward<U>(u));
135 }
136 }
137};

139// TODO(absl-team): Add another class using
140// std::is_trivially_move_constructible trait when available to match
141// http://cplusplus.github.io/LWG/lwg-defects.html#2900, for types that
142// have trivial move but nontrivial copy.
143// Also, we should be checking is_trivially_copyable here, which is not
144// supported now, so we use is_trivially_* traits instead.

158template <typename T>
159class optional_data<T, false> : public optional_data_base<T> {
160 protected:
161 using optional_data_base<T>::optional_data_base;
162
163 optional_data() = default;
164
165 optional_data(const optional_data& rhs) : optional_data_base<T>() {
166 if (rhs.engaged_) {
167 this->construct(rhs.data_);
168 }
169 }
170
171 optional_data(optional_data&& rhs) noexcept(
172 absl::default_allocator_is_nothrow::value ||
173 std::is_nothrow_move_constructible<T>::value)
174 : optional_data_base<T>() {
175 if (rhs.engaged_) {
176 this->construct(std::move(rhs.data_));
177 }
178 }
179
180 optional_data& operator=(const optional_data& rhs) {
181 if (rhs.engaged_) {
182 this->assign(rhs.data_);
183 } else {
184 this->destruct();
185 }
186 return *this;
187 }
188
189 optional_data& operator=(optional_data&& rhs) noexcept(
190 std::is_nothrow_move_assignable<T>::value&&
191 std::is_nothrow_move_constructible<T>::value) {
192 if (rhs.engaged_) {
193 this->assign(std::move(rhs.data_));
194 } else {
195 this->destruct();
196 }
197 return *this;
198 }
